50static void mdlInitializeSizes(SimStruct *S)



51{



52    int_T i, numOutputPorts;



53



54    ssSetNumSFcnParams(S, NUM_ARGS);



55    if (ssGetNumSFcnParams(S) != ssGetSFcnParamsCount(S))



56    {



57        sprintf(msg, "%d input args expected, %d passed", 



58            NUM_ARGS, ssGetSFcnParamsCount(S));



59        ssSetErrorStatus(S, msg);



60        return;



61    }



62



63    ssSetNumContStates(S, 0);



64    ssSetNumDiscStates(S, 0);



65    if( !ssSetNumInputPorts(S, 0) )return;



66



67    numOutputPorts = (int_T)mxGetN(CHANNEL_ARG);



68



69    if( !ssSetNumOutputPorts(S, numOutputPorts) )return;



70    for (i = 0; i < numOutputPorts; i++)



71    {



72        ssSetOutputPortWidth(S, i, 1);



73    }



74



75    ssSetSimStateCompliance( S, HAS_NO_SIM_STATE );



76



77    ssSetNumSampleTimes(S, 1);



78    ssSetNumRWork(S, NUM_R_WORKS);



79    ssSetNumIWork(S, NUM_I_WORKS);



80    ssSetNumPWork(S, NUM_P_WORKS);



81    ssSetNumModes(S, 0);



82    ssSetNumNonsampledZCs(S, 0);



83



84    for (i = 0; i < NUM_ARGS; i++)



85        ssSetSFcnParamTunable(S, i, 0);



86



87    ssSetOptions(S, SS_OPTION_DISALLOW_CONSTANT_SAMPLE_TIME | SS_OPTION_EXCEPTION_FREE_CODE );



88}

90static void mdlInitializeSampleTimes(SimStruct *S)



91{



92    ssSetModelReferenceSampleTimeInheritanceRule(S, USE_DEFAULT_FOR_DISCRETE_INHERITANCE);



93    if (mxGetPr(SAMP_TIME_ARG)[0] == -1.0)



94    {



95        ssSetSampleTime(S, 0, INHERITED_SAMPLE_TIME);



96        ssSetOffsetTime(S, 0, FIXED_IN_MINOR_STEP_OFFSET);



97    } else {



98        ssSetSampleTime(S, 0, mxGetPr(SAMP_TIME_ARG)[0]);



99        ssSetOffsetTime(S, 0, 0.0);



100    }



101}

104static void mdlStart(SimStruct *S)



105{



106#ifndef MATLAB_MEX_FILE



107    //int_T   *IWork   = ssGetIWork(S);



108    int_T    devtype  = (int_T) mxGetPr(DEV_ARG)[0];



109    int_T    bus, slot;



110    xpcPCIDevice pciinfo;



111    volatile int32_T *lreg;



112    volatile uint16_T *sreg;



113    const char *devName;



114    MseriesBoard *bd;



115    uint32_T pfidir = (uint32_T)mxGetPr(PFIDIR_ARG)[0];



116    int32_T bnum;



117    int32_T success;



118    uint32_T achan, bchan, zchan;



119    //int_T    i;



120    int_T    chan, reload, phase, filter;



121



122    if (mxGetN(SLOT_ARG) == 1)



123    {



124        bus = 0;



125        slot = (int_T)mxGetPr(SLOT_ARG)[0];



126    }



127    else



128    {



129        bus = (int_T)mxGetPr(SLOT_ARG)[0];



130        slot = (int_T)mxGetPr(SLOT_ARG)[1];



131    }



132



133    devtype = (int_T)mxGetPr(DEV_ARG)[0];



134



135    bd = niMIdentify( S, devtype, bus, slot, &pciinfo );



136    if( bd == 0 )



137        return;  // We didn't find the board as described.



138    devName = bd->name;



139    



140    lreg  = (volatile int32_T *)pciinfo.VirtAddress[1];



141    sreg  = (volatile uint16_T *)pciinfo.VirtAddress[1];



142



143    bnum = niM_Private_Data( (void *)sreg );



144    if( bnum < 0 )



145    {



146        sprintf(msg, "%s: Too many M series boards", devName );



147        ssSetErrorStatus(S, msg);



148        return;



149    }



150    ssSetPWorkValue(S, BASE_ADDR_P_IND, (void *)lreg);



151



152    chan = (int_T)mxGetPr(CHANNEL_ARG)[0];



153



154    niM_configureTimebase( (void *)sreg );



155    niM_Counter_Reset_All( (void *)sreg, chan );



156



157    sreg[ S_IO_Bidirection_Pin ] = pfidir;



158



159    filter = (int_T)mxGetPr(FILTER_ARG)[0] - 1;



160    // Set both encoder channels and the index pulse.



161    if( chan == 0 )



162    {



163        if( devtype == 27 )  // Special 6221/37



164        {



165            achan = 0;



166            bchan = 2;



167            zchan = 1;



168        }



169        else  // All the rest



170        {



171            achan = 8;



172            bchan = 10;



173            zchan = 9;



174        }



175    }



176    else



177    {



178        if( devtype == 27 )  // Special 6221/37



179        {



180            achan = 3;



181            bchan = 5;



182            zchan = 4;



183        }



184        else  // All the rest



185        {



186            achan = 3;



187            bchan = 11;



188            zchan = 4;



189        }



190    }



191    



192    success = niM_SetPFIFilter( bnum, achan, filter );



193    success = niM_SetPFIFilter( bnum, bchan, filter );



194    success = niM_SetPFIFilter( bnum, zchan, filter );



195



196    // Encoder setup



197    {



198        uint16_T mode;



199        uint16_T countmode;



200        uint16_T cmode;



201        int32_T initCount;



202        



203        sreg[ S_G0_Mode + chan ] = 0; // Point to LOAD Source A



204        // load initial value to both load registers



205        initCount = (int32_T)mxGetPr(INIT_COUNT_ARG)[0];



206



207        sreg[ S_G0_Command + chan ] = 0x0E40;  // Select bank X



208



209        // Duplicate write A and B



210        lreg[ L_G0_Load_A + 2*chan ] = initCount;



211



212        // sync gate, load counter



213        sreg[ S_G0_Command + chan ] = 0x0144;



214



215        lreg[ L_G0_Load_B + 2*chan ] = initCount;



216        lreg[ L_G0_Load_A + 2*chan ] = initCount;



217



218        // index phase



219        phase = (int_T)mxGetPr(INDEX_PHASE_ARG)[0] - 1;



220        // Gi_Alternate_Sync = 0x4000, only needed if 80 MHz source



221        // but it doesn't hurt for slower speeds.



222        countmode = 0x4000;



223        // Gi_Index_Phase = 0x0060 bits,



224        //    0 = a low, b low



225        //    1 = a low, b high



226        //    2 = a high, b low



227        //    3 = a high, b high



228        // Depends on when the encoder asserts index.



229        countmode |= (phase & 3) << 5;



230        cmode = (uint16_T)mxGetPr(COUNT_MODE_ARG)[0];



231        // Gi_Counting_Mode = 0x0007, bottom 3 bits



232        //    0 not an incremental encoder, normal mode



233        //    1 = x1 incremental



234        //    2 = x2 incremental



235        //    3 = x4 incremental



236        //    4 and 5 are not incremental encoder modes.



237        // Only 1, 2 or 3 can come from the block.



238        countmode |= cmode & 0x7;



239        reload = (int_T)mxGetPr(RELOAD_ARG)[0];  // 0 noreload, 1 reset on index



240        if( reload == 1 )



241            countmode |= 0x10;



242        sreg[ S_G0_MSeries_Counting_Mode + chan ] = countmode;



243



244        switch( chan )



245        {



246            // Use the input pins that are documented in the NI user



247            // manual for the M series boards.



248            case 0:



249                if( devtype == 27 )  // Special 6221/37



250                {



251                    // A = SRC = PFI0 = 1 << 10 = 0x01 << 10 = 0x0400



252                    // B = AUX = PFI2 = 3 << 5 = 0x03 << 5 = 0x0060



253                    // Z = Gate = PFI1 = 2 << 0 = 0x0002



254                    sreg[ S_G0_MSeries_ABZ ] = 0x0462;



255                }



256                else



257                {



258                    // A = SRC = PFI8 = 9 << 10 = 0x09 << 10 = 0x2400



259                    // B = AUX = PFI10 = 21 << 5 = 0x15 << 5 = 0x02a0



260                    // Z = Gate = PFI9 = 0xA << 0 = 0x000a



261                    sreg[ S_G0_MSeries_ABZ ] = 0x26aa;



262                }



263                break;



264            case 1:



265                if( devtype == 27 )  // Special 6221/37



266                {



267                    // A = SRC = PFI3 = 4 << 10 = 0x04 << 10 = 0x1000



268                    // B = AUX = PFI5 = 6 << 5 = 0x06 << 5 = 0x00c0



269                    // Z = Gate = PFI4 = 5 << 0 = 0x0005



270                    sreg[ S_G1_MSeries_ABZ ] = 0x10c5;



271                }



272                else



273                {



274                    // A = SRC = PFI3 = 4 << 10 = 0x1000



275                    // B = AUX = PFI11 = 22 << 5 = 0x16 << 5 = 0x02c0



276                    // Z = Gate = PFI4 = 5 << 0 = 0x0005



277                    sreg[ S_G1_MSeries_ABZ ] = 0x12c5;



278                }



279                break;



280        }



281        



282        // no gated start/stop, rising edge gate



283        // Load source select(A) = 0 << 7



284        // 



285        mode = 0x119;



286        if( reload == 1 )



287            mode |= 0x4000;  // Reload on gate, index pulse to gate.



288        sreg[ S_G0_Mode + chan ] = mode;



289



290        // Synchronous gate, aux (B) direction, armed, following value



291        sreg[ S_G0_Command + chan ] = 0x0141;



292    }



293#endif



294}

296static void mdlOutputs(SimStruct *S, int_T tid)



297{



298#ifndef MATLAB_MEX_FILE



299    int_T   chan  = (int_T)mxGetPr(CHANNEL_ARG)[0];



300    real_T *y     = ssGetOutputPortRealSignal(S, 0);



301    volatile int32_T *lreg = (volatile int32_T *)ssGetPWorkValue(S, BASE_ADDR_P_IND);



302    volatile uint16_T *sreg = (volatile uint16_T *)lreg;



303    int32_T value;



304    



305    // synch gate, counting direction = aux pin 



306    sreg[ S_G0_Command + chan ] = 0x0140;



307    



308    // synch gate, counting direction = aux pin, save trace



309    sreg[ S_G0_Command + chan ] = 0x0142;



310



311    // read count



312    value = lreg[L_G0_Save + chan];



313    y[0] = (real_T)value;



314#endif



315}

317static void mdlTerminate(SimStruct *S)



318{



319#ifndef MATLAB_MEX_FILE



320    int_T   chan  = (int_T)mxGetPr(CHANNEL_ARG)[0];



321    volatile uint16_T *sreg = (volatile uint16_T *)ssGetPWorkValue(S, BASE_ADDR_P_IND);



322



323    sreg[ S_G0_Command + chan ] = 0x0010;  // Disarm on stop



324#endif



325}
