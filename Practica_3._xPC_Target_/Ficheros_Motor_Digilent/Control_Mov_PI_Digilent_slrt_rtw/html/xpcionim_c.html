33void niM_AISetup( void *base,



34                  bool reset5x,



35                  uint32_T *channels,



36                  uint32_T *range,



37                  uint32_T *coupling,

53    sreg[S_Joint_Reset] = 1;  // do the reset

55



56    dummy = sreg[S_Joint_Status_1 ];  // else Watcom combines writes!



57



58    sreg[S_Interrupt_A_Enable] = 0;  // None enabled



59    sreg[S_Interrupt_A_Ack] = 0x3f8; // all ints cleared



60    sreg[S_AI_Mode_1] = 0x0008;



61    dummy = sreg[S_Joint_Status_1 ];  // else Watcom combines writes!



62    sreg[S_AI_Mode_2] = 0x0000;



63    sreg[S_AI_Mode_3] = 0x0000;



64



65    sreg[S_Joint_Reset] = 0x100;  // Configuration end



66    dummy = sreg[S_Joint_Status_1 ];  // else Watcom combines writes!



67    sreg[S_Joint_Reset] = 0x10;  // configuration mode, no glitches



68



69    if( reset5x )  // 5x boards



70        sreg[S_AI_Output_Control] = 0x0002; // kAI_CONVERT_Output_SelectActive_Low=2



71    else  // all the rest



72        sreg[S_AI_Output_Control] = 0x0000;



73



74    sreg[S_AI_Personal] = 0x0000;



75



76    sreg[S_Joint_Reset] = 0x100;  // Configuration end



77    dummy = sreg[S_Joint_Status_1 ];  // else Watcom combines writes!



78    sreg[S_Joint_Reset] = 0x10;  // configuration mode, no glitches



79



80    sreg[S_AI_START_STOP_Select] = 0x0000;



81    dummy = sreg[S_Joint_Status_1 ];  // else Watcom combines writes!



82    sreg[S_AI_Trigger_Select] = 0x0000;



83



84    if( reset5x )  // 5x boards



85        sreg[S_AI_Output_Control] = 0x302;



86    else  // all the rest



87        sreg[S_AI_Output_Control] = 0x303;



88    sreg[S_AI_Personal] = 0x0400;



89



90    sreg[S_AI_START_STOP_Select] = 0x0000;



91    dummy = sreg[S_Joint_Status_1 ];  // else Watcom combines writes!



92    sreg[S_AI_Trigger_Select] = 0x0000;



93



94    sreg[S_AI_FIFO_Clear] = 1;



95



96    if( reset5x )



97    {



98        // NOTE: the correct order is write 0, then 1, per a comment



99        // on the NI website in the DDK message board from 11-21-2005.



100        //board->Static_AI_Control[0].writeRegister (0);



101        breg[B_Static_AI_Control0] = 0;



102        //board->Static_AI_Control[0].writeRegister (1);



103        breg[B_Static_AI_Control0] = 1;



104    



105        //board->AI_Command_1.writeAI_CONVERT_Pulse (1);



106        sreg[S_AI_Command_1] = 1;  // strobe bit, resets



107        //board->AI_Command_1.writeAI_CONVERT_Pulse (1);



108        sreg[S_AI_Command_1] = 1;



109        //board->AI_Command_1.writeAI_CONVERT_Pulse (1);



110        sreg[S_AI_Command_1] = 1;



111    }



112



113    sreg[S_Joint_Reset] = 0x0001;  // Disarm and reset AI registers



114



115    sreg[S_Configuration_Memory_Clear] = 1; // clear config memory



116    lreg[L_AI_Config_FIFO_Bypass] = 0;



117



118    for( idx = 0 ; idx < nchans ; idx++ )



119    {



120        niM_aiConfigureChannel( base,



121                                (uint16_T)channels[idx],



122                                (uint16_T)range[idx],



123                                (uint16_T)polarity[idx],  /* bipolar == 0 */



124                                (uint16_T)coupling[idx],



125                                (bool)((idx == (nchans-1)) ? true : false ));



126    }



127



128    sreg[ S_AI_Mode_3 ] = 0;  // set FIFO request mode



129



130    // aiTrigger



131    sreg[ S_AI_Mode_1 ] = 0x9;  // require a software write for each scan



132    sreg[ S_AI_Trigger_Select ] = START1_SELECT(0)



133                                | START1_EDGE(1)



134                                | START1_SYNC(1)



135                                | START1_POLARITY(0)



136                                | START2_SELECT(0)



137                                | START2_EDGE(1)



138                                | START2_SYNC(1)



139                                | START2_POLARITY(0);



140    sreg[ S_AI_Mode_2 ] = 0;



141



142    // aiSampleStop



143    // AI_Stop_SelectIN = 0x13 << 7 = 0x0980



144    // AI_STOP_PolarityActive_High = 0



145    // AI_STOP_Sync = 1 << 13 = 0x2000



146    // AI_STOP_Edge = 1 << 12 = 0x1000, if multichannel



147    ststop = 0x2980 | ((nchans > 1) ? 0x1000 : 0 );



148    sreg[ S_AI_START_STOP_Select ] = ststop;



149    



150    // aiNumberOfSamples, number of scans through the channel list



151    // original: sreg[ S_AI_Mode_1 ] = 0x9 | 0x2;  // AI_Continuous | AI_Trigger_Once;



152    sreg[ S_AI_Mode_1 ] = 0x9 | 0x2;  // AI_Continuous | AI_Trigger_Once



153    lreg[ L_AI_SC_Load_A ] = 0; // postTriggerSamples-1;



154    lreg[ L_AI_SC_Load_B ] = 0;



155    sreg[ S_AI_Command_1 ] = 0x20;  // AI_SC_Load, strobe bit, resets



156



157    // aiSampleStart



158    // AI_START_SelectPulse = 0x12



159    // AI_START_PolarityRisingEdge = 0 in bit 15



160    // AI_START_Sync = 1 << 6 = 0x40



161    // AI_START_Edge = 1 << 5 = 0x20



162    ststop |= 0x12 | 0x60;



163    sreg[ S_AI_START_STOP_Select ] = ststop;



164



165    // aiConvert



166    sreg[ S_AI_Mode_1 ] = 0xb;  // Sum of all bits we already set.



167    dummy = sreg[S_Joint_Status_1 ];  // else Watcom combines writes!



168



169    sreg[ S_AI_Mode_2 ] = 0x100;



170    sreg[ S_AI_Mode_3 ] = 0; // given externalSampleClock == false



171    lreg[ L_AI_SI2_Load_A ] = 10; // scancount;  // delayDivisor;, use a minimum val?



172    lreg[ L_AI_SI2_Load_B ] = scancount;  // periodDivisor;



173    sreg[ S_AI_Mode_2 ] = 0x100;     // Load A, initial delay



174    sreg[ S_AI_Command_1 ] = 0x800;  // Strobe bit, resets



175    sreg[ S_AI_Mode_2 ] = 0x300;     // Load B, sample separation



176



177    sreg[S_AI_FIFO_Clear] = 1;



178



179    sreg[S_Joint_Reset] = 0x100;  // Configuration end



180



181    // aiArm



182    sreg[ S_AI_Command_1 ] = 0x1040;



183



184    // aiStart



185    sreg[ S_AI_Command_2 ] = 0x0001;



186}

323MseriesBoard *niMIdentify( SimStruct *S, int deviceType, int bus, int slot, xpcPCIDevice *pciinfo )



324{



325    int i;



326    int j;



327    const char *devName;



328    uint32_T devId;



329    uint32_T found = 0;



330



331    devId = 0;



332    for( i = 0 ; i < sizeof(SupportedBoard)/sizeof(MseriesBoard) ; i++ )



333    {



334        if( SupportedBoard[i].Type == (uint32_T)deviceType )



335        {



336            for( j = 0 ; j < 4 ; j++ )



337            {



338                devId = SupportedBoard[i].devID[j];



339                if( devId != 0 )



340                {



341                    // return 1 if getinfo error



342                    // return 0 if ok



343                    if( niMconfig( devId, bus, slot, pciinfo ) != 1 )



344                    {



345                        found = 1;



346                        break;



347                    }



348                }



349            }



350            if( found == 0 ) // devtype found, but board isn't present



351            {



352                devName = SupportedBoard[i].name;



353                sprintf(msg,"%s (bus %d, slot %d): board not present", devName, bus, slot );



354                ssSetErrorStatus(S,msg);



355                return 0;



356            }



357



358            return &SupportedBoard[i];



359       }



360    }



361    sprintf( msg, "NI M series: Unknown board type %d", deviceType );



362    ssSetErrorStatus(S,msg);



363    return 0;



364}

366// ----- PCI configuration subroutine --------



367int niMconfig( int deviceId, int bus, int slot, xpcPCIDevice *pciinfo )



368{



369    // return 1 if get info error



370    // return 0 if ok



371    uint16_T  vendorId;



372    uint16_T  subvendorId, subdeviceId;



373    volatile unsigned int *lbar0;



374



375    vendorId = 0x1093;



376    subvendorId = XPC_NO_SUB;  // Not checked



377    subdeviceId = XPC_NO_SUB;  // Not checked



378



379    // look for the PCI-Device



380    if( xpcGetPCIDeviceInfo( vendorId, (uint16_T)deviceId, subvendorId, subdeviceId, bus, slot, pciinfo ) )



381    {



382        return 1;



383    }



384



385    // The MITE chip



386    pciinfo->VirtAddress[0] =



387        (unsigned long)xpcReserveMemoryRegion( (void *)pciinfo->BaseAddress[0],



388                                               4096,



389                                               XPC_RT_PG_USERREADWRITE );



390    // Register space for IO



391    pciinfo->VirtAddress[1] =



392        (unsigned long)xpcReserveMemoryRegion( (void *)pciinfo->BaseAddress[1],



393                                               4096,



394                                               XPC_RT_PG_USERREADWRITE );



395



396    // show Device Information



397    //xpcShowPCIDeviceInfo( pciinfo );



398



399    // We need to get to register space as long, short or byte.



400    // For 32 bit wide registers: lreg[L_REGNAME]



401    // For 16 bit wide registers: sreg[S_REGNAME]



402    // For 8 bit wide registers: breg[B_REGNAME]



403    // The register names indicate byte, short or long.



404



405    // Enable the MITE



406    lbar0 = (unsigned int *)pciinfo->VirtAddress[0];



407    lbar0[48] = ((unsigned int)pciinfo->BaseAddress[1] & 0xffffff00) | 0x00000080;



408



409    return 0;  // no error



410    // --- end common PCI config subroutine ----



411}

423uint32_T niM_Private_Data( void *base )



424{



425    static int firsttime = 1;



426    int i;



427    



428    if( firsttime == 1 )



429    {



430        for( i = 0 ; i < NIM_MAXBOARDS ; i++ )



431        {



432            private_data[i].base = 0;



433            private_data[i].command[0] = 0;



434            private_data[i].command[1] = 0;



435            private_data[i].pfi_config[0] = 0;  // All default PFI



436            private_data[i].pfi_config[1] = 0;



437            private_data[i].pfi_config[2] = 0;



438            private_data[i].pfi_config[3] = 0;



439            private_data[i].pfi_config[4] = 0;



440            private_data[i].pfi_config[5] = 0;



441            private_data[i].pfi_filter = 0;



442            private_data[i].pfi_do = 0;



443        }



444        firsttime = 0;  // Don't repeat the initialization



445    }



446    // Look for this base address, return index if found



447    for( i = 0 ; i < NIM_MAXBOARDS ; i++ )



448    {



449        if( base == private_data[i].base ) // found



450            return i;



451    }



452    // Didn't find it, look for the first 0 base, claim it.



453    for( i = 0 ; i < NIM_MAXBOARDS ; i++ )



454    {



455        if( private_data[i].base == 0 )



456        {



457            private_data[i].base = base;



458            return i;



459        }



460    }



461    return -1;  // Didn't find it and no more room.



462}

464int32_T niM_SetPFIFilter( uint32_T bnum, uint32_T chan, uint32_T value )



465{



466    // chan is the PFI number, [0, 15]



467    // value is in [0, 1, 2, 3]



468    uint32_T mask = 3 << (2*chan);



469    uint32_T filt = private_data[bnum].pfi_filter;



470    volatile uint32_T *lreg = (volatile uint32_T *)private_data[bnum].base;



471



472    if( lreg == 0 )



473        return -1;  // Null pointer, bad bnum



474    



475    if( value > 3 )



476        return -2;  // error, out of range



477    



478    if( chan > 15 )



479        return -3; // error, channel out of range



480    



481    filt = (filt & ~mask) | ((value & 3) << (2*chan));



482    private_data[bnum].pfi_filter = filt;



483    lreg[ L_PFI_Filter ] = filt;



484    return 0;



485}

487// Set connect to 0 to get the default connection, or



488// Set it to the value from the NI doc to choose a non-default connection



489void niM_SetPFI_Select( uint32_T bnum, uint32_T pfinum, uint32_T connect )



490{



491    uint32_T idx = pfinum/3;



492    uint32_T shift = (pfinum % 3) * 5;



493    uint32_T val = private_data[bnum].pfi_config[idx];



494    volatile uint16_T *sreg = (volatile uint16_T *)private_data[bnum].base;



495    



496    connect &= 0x1f;



497    val = (val & ~(0x1f << shift)) | (connect << shift);



498    



499    private_data[bnum].pfi_config[idx] = val;



500    sreg[S_PFI_Output_Select_1 + idx] = val;



501}

503// Get the previous DO value, add the specified bit and return it.



504// The caller writes it to the register.



505uint16_T niM_SetPFI_DO( uint32_T bnum, uint32_T pfinum, uint32_T val )



506{



507    uint16_T doval = private_data[bnum].pfi_do;



508    doval = (doval & ~(1 << pfinum)) | ((val & 1) << pfinum);



509    private_data[bnum].pfi_do = doval;



510    return doval;



511}

513/*



514 * When using the G0 and G1 register names, simply adding



515 * 1 to the G0 define gives the correct value for the G1



516 * define.



517 * That works for both the short and long registers.



518 * L_G0_Load_A +1 == L_G0_Load_B since the byte addresses



519 * differ by 4, but the symbols divide by 4 to get a word offset.
